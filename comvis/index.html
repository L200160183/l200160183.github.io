import time
import cv2 as cv

def brighness(image, brighness):
    if image is None:
        print('Could not open or find the image: ', image)
        exit(0)
    alpha = 1.0  # Simple contrast control
    beta = brighness  # Simple brightness control
    new_image = cv.convertScaleAbs(image, alpha=alpha, beta=beta)
    return new_image
